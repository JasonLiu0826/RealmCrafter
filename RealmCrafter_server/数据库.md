### 🗄️ RealmCrafter 完整数据库 SQL 脚本

SQL

```sql
-- ==========================================================
-- RealmCrafter 企业级数据库终极初始化脚本 (MySQL 8.0+)
-- 整合内容：UGC生态、虚拟币分润、IM通讯、合规风控、版本防覆盖、AI对账
-- ==========================================================

CREATE DATABASE IF NOT EXISTS `realmcrafter` DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE `realmcrafter`;

-- --------------------------------------------------------
-- 1. 用户核心表 (user)
-- --------------------------------------------------------
CREATE TABLE `user` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '用户账号ID(前3位+中4位+后4位)',
  `username` VARCHAR(64) NOT NULL COMMENT '用户名/昵称',
  `password` VARCHAR(255) NOT NULL COMMENT '密码哈希值',
  `role` ENUM('USER', 'ADMIN', 'SUPER_ADMIN') NOT NULL DEFAULT 'USER' COMMENT '角色',
  `avatar` VARCHAR(512) DEFAULT NULL COMMENT '头像URL',
  `gender` ENUM('MALE', 'FEMALE', 'UNKNOWN') DEFAULT 'UNKNOWN' COMMENT '性别',
  `age` TINYINT UNSIGNED DEFAULT NULL COMMENT '年龄',
  `signature` VARCHAR(255) DEFAULT NULL COMMENT '个人签名',
  `level` INT NOT NULL DEFAULT 1 COMMENT '贡献评级(1-5级)',
  `current_theme_id` VARCHAR(32) DEFAULT 'classic_white' COMMENT '当前选中的主题ID(实现多端同步)',
  `token_balance` BIGINT NOT NULL DEFAULT 100000 COMMENT 'AI算力余额(默认送10万Token)',
  `crystal_balance` DECIMAL(10,2) NOT NULL DEFAULT '0.00' COMMENT '虚拟币余额(灵能水晶,用于广场交易)',
  `vip_expire_time` DATETIME DEFAULT NULL COMMENT 'VIP到期时间',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_username` (`username`)
) ENGINE=InnoDB AUTO_INCREMENT=10010001000 DEFAULT CHARSET=utf8mb4 COMMENT='用户账号表';

-- --------------------------------------------------------
-- 2. 资金流水对账表 (wallet_transaction)
-- --------------------------------------------------------
CREATE TABLE `wallet_transaction` (
  `id` BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `user_id` BIGINT NOT NULL COMMENT '用户ID',
  `amount` DECIMAL(10,2) NOT NULL COMMENT '变动水晶数量(正负值)',
  `type` ENUM('RECHARGE', 'BUY_ASSET', 'CREATOR_REVENUE', 'PENALTY_DEDUCT') NOT NULL COMMENT '充值/购买/创作者收益/违规扣罚',
  `description` VARCHAR(255) DEFAULT NULL COMMENT '账单明细描述',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  KEY `idx_user_id` (`user_id`),
  CONSTRAINT `fk_wallet_user` FOREIGN KEY (`user_id`) REFERENCES `user` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='虚拟币流水表';

-- --------------------------------------------------------
-- 3. 设定集表 (setting_pack)
-- --------------------------------------------------------
CREATE TABLE `setting_pack` (
  `id` VARCHAR(32) NOT NULL PRIMARY KEY COMMENT 'sd+用户ID后4位+8位随机数',
  `user_id` BIGINT NOT NULL COMMENT '创作者ID',
  `source_setting_id` VARCHAR(32) DEFAULT NULL COMMENT '血统追踪:源设定集ID(用于Fork识别)',
  `title` VARCHAR(128) NOT NULL COMMENT '标题',
  `cover` VARCHAR(512) DEFAULT NULL COMMENT '封面',
  `description` VARCHAR(512) DEFAULT NULL COMMENT '简介',
  `content` JSON NOT NULL COMMENT '五大维度JSON数据',
  `price` DECIMAL(10,2) NOT NULL DEFAULT '0.00' COMMENT '定价(水晶)',
  `allow_modify` TINYINT(1) NOT NULL DEFAULT 1 COMMENT '是否允许Fork后修改',
  `is_public` TINYINT(1) NOT NULL DEFAULT 1 COMMENT '是否发布到广场',
  `status` ENUM('NORMAL', 'PENDING_REVIEW', 'FROZEN_COPYRIGHT') NOT NULL DEFAULT 'NORMAL' COMMENT '状态(正常/机审中/版权冻结)',
  `version_id` BIGINT NOT NULL DEFAULT 1 COMMENT '乐观锁版本号(防同步覆盖)',
  `device_hash` VARCHAR(64) DEFAULT NULL COMMENT '最后修改设备指纹',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  KEY `idx_user_id` (`user_id`),
  CONSTRAINT `fk_setting_user` FOREIGN KEY (`user_id`) REFERENCES `user` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='设定集图纸表';

-- --------------------------------------------------------
-- 4. 故事包表 (story)
-- --------------------------------------------------------
CREATE TABLE `story` (
  `id` VARCHAR(32) NOT NULL PRIMARY KEY COMMENT 'gs+用户ID后4位+8位随机数',
  `user_id` BIGINT NOT NULL COMMENT '所有者ID',
  `setting_pack_id` VARCHAR(32) NOT NULL COMMENT '关联设定集',
  `source_story_id` VARCHAR(32) DEFAULT NULL COMMENT '血统追踪:源故事ID',
  `title` VARCHAR(128) NOT NULL,
  `cover` VARCHAR(512) DEFAULT NULL,
  `description` VARCHAR(512) DEFAULT NULL,
  `price` DECIMAL(10,2) NOT NULL DEFAULT '0.00' COMMENT '定价(水晶)',
  `allow_download` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '允许克隆下载',
  `clone_includes_settings` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '克隆是否带设定修改权',
  `is_public` TINYINT(1) NOT NULL DEFAULT 1,
  `status` ENUM('NORMAL', 'PENDING_REVIEW', 'FROZEN_COPYRIGHT') NOT NULL DEFAULT 'NORMAL',
  `version_id` BIGINT NOT NULL DEFAULT 1 COMMENT '乐观锁版本号',
  `device_hash` VARCHAR(64) DEFAULT NULL,
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  KEY `idx_user_id` (`user_id`),
  CONSTRAINT `fk_story_user` FOREIGN KEY (`user_id`) REFERENCES `user` (`id`) ON DELETE CASCADE,
  CONSTRAINT `fk_story_setting` FOREIGN KEY (`setting_pack_id`) REFERENCES `setting_pack` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='互动故事表';

-- --------------------------------------------------------
-- 5. 章节内容表 (chapter)
-- --------------------------------------------------------
CREATE TABLE `chapter` (
  `id` BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `story_id` VARCHAR(32) NOT NULL,
  `chapter_index` INT NOT NULL COMMENT '章节序号',
  `title` VARCHAR(255) DEFAULT NULL,
  `content` TEXT NOT NULL COMMENT '净化后的正文',
  `branches_data` JSON NOT NULL COMMENT 'AI生成的JSON分支数据',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_story_index` (`story_id`, `chapter_index`),
  CONSTRAINT `fk_chapter_story` FOREIGN KEY (`story_id`) REFERENCES `story` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='故事章节记录';

-- --------------------------------------------------------
-- 6. 版权仲裁工单表 (copyright_report)
-- --------------------------------------------------------
CREATE TABLE `copyright_report` (
  `id` BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `reporter_id` BIGINT NOT NULL COMMENT '举报人ID',
  `target_type` ENUM('STORY', 'SETTING') NOT NULL COMMENT '资产类型',
  `target_id` VARCHAR(32) NOT NULL COMMENT '资产ID',
  `reason` VARCHAR(500) NOT NULL COMMENT '举报理由/调色盘比对',
  `status` ENUM('PENDING', 'ACCEPTED', 'REJECTED') NOT NULL DEFAULT 'PENDING' COMMENT '处理状态',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT `fk_report_user` FOREIGN KEY (`reporter_id`) REFERENCES `user` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='版权仲裁表';

-- --------------------------------------------------------
-- 7. 社交关系/互动表
-- --------------------------------------------------------
CREATE TABLE `user_follow` (
  `id` BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `follower_id` BIGINT NOT NULL COMMENT '粉丝',
  `following_id` BIGINT NOT NULL COMMENT '被关注者',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_follow` (`follower_id`, `following_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户关注表';

CREATE TABLE `user_interaction` (
  `id` BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `user_id` BIGINT NOT NULL,
  `target_type` ENUM('STORY', 'SETTING', 'COMMENT') NOT NULL,
  `target_id` VARCHAR(32) NOT NULL,
  `action_type` ENUM('LIKE', 'FAVORITE', 'FORK') NOT NULL,
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_interaction` (`user_id`, `action_type`, `target_type`, `target_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户交互记录(赞/藏/拉取)';

-- --------------------------------------------------------
-- 8. 系统消息与通知表
-- --------------------------------------------------------
CREATE TABLE `system_notification` (
  `id` BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `user_id` BIGINT NOT NULL,
  `title` VARCHAR(128) NOT NULL,
  `content` VARCHAR(512) NOT NULL,
  `type` ENUM('SYSTEM', 'COPYRIGHT', 'REWARD', 'SECURITY') NOT NULL,
  `is_read` TINYINT(1) NOT NULL DEFAULT 0,
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统通知表';

-- --------------------------------------------------------
-- 9. AI 算力对账流水表 (ai_usage_log)
-- --------------------------------------------------------
CREATE TABLE `ai_usage_log` (
  `id` BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `user_id` BIGINT NOT NULL,
  `story_id` VARCHAR(32) DEFAULT NULL,
  `model_name` VARCHAR(64) NOT NULL COMMENT '调用模型名称',
  `prompt_tokens` INT NOT NULL,
  `completion_tokens` INT NOT NULL,
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  KEY `idx_user_time` (`user_id`, `create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='AI Token消耗审计表';

-- --------------------------------------------------------
-- 10. IM 聊天消息表 (message)
-- --------------------------------------------------------
CREATE TABLE `message` (
  `id` BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `sender_id` BIGINT NOT NULL,
  `receiver_id` BIGINT NOT NULL,
  `msg_type` ENUM('TEXT', 'IMAGE', 'ASSET_CARD') NOT NULL,
  `payload` JSON NOT NULL COMMENT '富文本消息载荷',
  `is_read` TINYINT(1) NOT NULL DEFAULT 0,
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  KEY `idx_chat` (`sender_id`, `receiver_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='IM聊天消息表';
```
